letcurrentToken=null;
asyncfunctionapi(endpoint,method='GET',data=null,auth=true){
constheaders={'Content-Type':'application/json'};
if(auth&&currentToken)headers['Authorization']=currentToken;
constopts={method,headers};
if(data)opts.body=JSON.stringify(data);
constres=awaitfetch(/api/${endpoint},opts);
constct=res.headers.get('content-type');
if(!ct||!ct.includes('application/json'))thrownewError('Invalidresponse');
constjson=awaitres.json();
if(!res.ok)thrownewError(json.error||'请求失败');
returnjson;
}
asyncfunctionapiUpload(endpoint,file,extra={}){
constfd=newFormData();
fd.append('file',file);
Object.entries(extra).forEach(([k,v])=>fd.append(k,v));
constheaders={};
if(currentToken)headers['Authorization']=currentToken;
constres=awaitfetch(/api/${endpoint},{method:'POST',headers,body:fd});
constjson=awaitres.json();
if(!res.ok)thrownewError(json.error||'上传失败');
returnjson;
}

一、自适应设计哲学
1.1核心原则
|原则|说明|
|------|------|
|Content-First|内容决定断点，而非设备决定断点|
|Fluid-by-Default|所有可流动的属性默认使用流体值，断点仅作修正|
|ProgressiveEnhancement|从最小屏幕向上增强，基础体验不依赖任何高级特性|
|Container-Aware|组件响应自身容器尺寸，而非视口尺寸|
|Input-Adaptive|同一断点下区分触控与指针设备|
|Density-Aware|感知像素密度，调整资源与细节层级|

3.1CSSClamp流体公式
所有字号使用clamp()函数实现在断点之间的平滑过渡，避免突变：
clamp(最小值,首选值（流体计算）,最大值)
首选值公式：
preferred=min-size+(max-size-min-size)×((100vw-min-viewport)/(max-viewport-min-viewport))
简化为CSS：
calc(min-size+(max-size-min-size)((100vw-min-vw)/(max-vw-min-vw)))

functiondetectPerformanceTier(){
constcanvas=document.createElement('canvas');

constdebugInfo=gl?.getExtension('WEBGL_debug_renderer_info');
constrenderer=debugInfo
?gl.getParameter(debugInfo.UNMASKED_RENDERER_WEBGL)
:'';
constmemory=navigator.deviceMemory||4;//GB
constcores=navigator.hardwareConcurrency||4;
constconnection=navigator.connection?.effectiveType||'4g';
letscore=0;
//内存评分
if(memory>=8)score+=3;
elseif(memory>=4)score+=2;
elsescore+=1;
//CPU核心评分
if(cores>=8)score+=3;
elseif(cores>=4)score+=2;
elsescore+=1;
//网络评分

elsescore+=0;

三十九、设计原则总结（扩展版）
1.一致性—所有交互反馈使用同一套弹性缓动函数，不因设备差异打破体验连贯性
2.层次感—通过透明度、模糊、阴影建立明确的视觉层级，并随设备能力分级降级
3.呼吸感—间距随视口流体伸缩，确保各尺寸下都有充裕的留白
4.柔和性—大圆角、柔和渐变、低对比度阴影在所有尺寸下保持一致调性
5.双模式—亮色/暗色在每个断点都有完整定义，无遗漏
6.反馈感—触控用:active，指针用:hover，键盘用:focus-visible，三通道完备
7.渐进增强—核心功能在180px手表屏上可用，装饰效果在高端设备上绽放
8.性能优先—自动检测设备性能并降级特效，确保60fps流畅体验
9.容器感知—组件根据自身容器而非视口响应，支持灵活嵌入任意布局
10.输入感知—触控与指针设备提供差异化的交互尺寸与反馈
11.安全区域—全面适配刘海屏、灵动岛、手势导航条、折叠屏铰链
12.可访问性—满足WCAG2.1AA标准，支持高对比度、减少动画、屏幕阅读器
13.国际化—支持RTL布局、CJK文字特性，使用逻辑属性，根据IP属地自动切换正确语言，另外允许用户自行设计
14.密度感知—根据DPR调整资源、边框粗细、细节层级
15.未来兼容—使用特性检测@supports保障向后兼容，同时拥抱新API
